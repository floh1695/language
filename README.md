## samples

### fizzbuzz

```
let a: 0
let b: 100
while a <= b {
    if a % 15 {
        print 'fizzbuzz'
    }
    elif a % 3 {
        print 'fizz'
    }
    elif a % 5 {
        print 'buzz'
    }
    else {
        print a
    }
    a: a + 1
}
```
